练习6: 简单利息计算

问题：编写一个程序，计算简单利息。用户输入本金(principal)、年利率(rate)和时间(years)，程序计算并显示利息金额。使用公式：利息 = 本金 × 利率 × 时间。

查看参考解答 
# 简单利息计算
try:
    principal = float(input("请输入本金: "))
    rate = float(input("请输入年利率(例如，5%输入0.05): "))
    years = float(input("请输入年数: "))
    # 计算简单利息
    interest = principal * rate * years
    # 显示结果
    print(f"\n简单利息计算结果:")
    print(f"本金: ${principal:,.2f}")
    print(f"年利率: {rate:.2%}")
    print(f"时间: {years}年")
    print(f"利息金额: ${interest:,.2f}")
    print(f"总金额: ${principal + interest:,.2f}")
except ValueError:
    print("错误：请输入有效的数字")

练习8: 投资回报率(ROI)计算

问题：编写一个程序，计算投资回报率(ROI)。用户输入初始投资金额和最终投资价值，程序计算并显示ROI。使用公式：ROI = (最终价值 - 初始投资) / 初始投资 × 100%。

查看参考解答 
# 投资回报率计算
try:
    initial_investment = float(input("请输入初始投资金额: "))
    final_value = float(input("请输入最终投资价值: "))
    # 计算ROI
    roi = (final_value - initial_investment) / initial_investment * 100
    # 显示结果
    print(f"\n投资回报率计算结果:")
    print(f"初始投资: ${initial_investment:,.2f}")
    print(f"最终价值: ${final_value:,.2f}")
    print(f"投资回报率: {roi:.2f}%")
except ValueError:
    print("错误：请输入有效的数字")
except ZeroDivisionError:
    print("错误：初始投资不能为零")

练习9: 贷款负担率计算

问题：编写一个程序，计算贷款负担率。用户输入月收入和月贷款支付金额，程序计算并显示贷款负担率。使用公式：贷款负担率 = 月贷款支付金额 / 月收入 × 100%。

查看参考解答 
# 贷款负担率计算
try:
    monthly_income = float(input("请输入月收入: "))
    monthly_payment = float(input("请输入月贷款支付金额: "))
    # 计算贷款负担率
    debt_burden_ratio = (monthly_payment / monthly_income) * 100
    # 显示结果
    print(f"\n贷款负担率计算结果:")
    print(f"月收入: ${monthly_income:,.2f}")
    print(f"月贷款支付: ${monthly_payment:,.2f}")
    print(f"贷款负担率: {debt_burden_ratio:.2f}%")
    # 提供建议
    if debt_burden_ratio > 40:
        print("警告：贷款负担率过高，建议控制在40%以下")
    elif debt_burden_ratio > 30:
        print("注意：贷款负担率略高，建议注意控制")
    else:
        print("贷款负担率在合理范围内")
except ValueError:
    print("错误：请输入有效的数字")
except ZeroDivisionError:
    print("错误：月收入不能为零")

练习10: 复利计算与格式化输出

问题：编写一个程序，计算复利并生成一个格式化的投资报告。用户输入本金、年利率和投资年限，程序计算每年末的投资价值并生成一个表格形式的报告。

查看参考解答 
# 复利计算与报告生成
try:
    principal = float(input("请输入本金: "))
    rate = float(input("请输入年利率(例如，5%输入0.05): "))
    years = int(input("请输入投资年限: "))
    # 打印表头
    print("\n年度投资报告")
    print("=" * 50)
    print(f"{'年度':^6}{'期初金额':>15}{'利息':>12}{'期末金额':>15}")
    print("-" * 50)
    # 计算并打印每年的数据
    current_amount = principal
    for year in range(1, years + 1):
        interest = current_amount * rate
        end_amount = current_amount + interest
        print(f"{year:^6}${current_amount:>14,.2f}${interest:>11,.2f}${end_amount:>14,.2f}")
        current_amount = end_amount
    print("=" * 50)
    # 计算总收益
    total_interest = end_amount - principal
    print(f"\n投资总结:")
    print(f"初始投资: ${principal:,.2f}")
    print(f"总收益: ${total_interest:,.2f}")
    print(f"最终金额: ${end_amount:,.2f}")
    print(f"收益率: {(total_interest/principal)*100:.2f}%")
except ValueError:
    print("错误：请输入有效的数字")
except ZeroDivisionError:
    print("错误：本金不能为零")

挑战练习

挑战1: 投资组合价值计算器

问题：创建一个程序，允许用户输入多个股票的信息（股票代码、持有数量和当前价格），计算投资组合的总价值，并生成一个格式化的投资组合报告。

查看参考解答 
# 投资组合价值计算器
def get_stock_info():
    stocks = []
    while True:
        try:
            print("\n输入股票信息 (输入空白股票代码结束)")
            symbol = input("请输入股票代码: ").upper()
            if not symbol:
                break
            shares = float(input("请输入持有数量: "))
            price = float(input("请输入当前价格: "))
            stocks.append({
                "symbol": symbol,
                "shares": shares,
                "price": price,
                "value": shares * price
            })
        except ValueError:
            print("错误：请输入有效的数字")
            continue
    return stocks
def print_portfolio_report(stocks):
    if not stocks:
        print("投资组合为空")
        return
    total_value = sum(stock["value"] for stock in stocks)
    print("\n投资组合报告")
    print("=" * 55)
    print(f"{'股票代码':^10}{'持有数量':>12}{'当前价格':>12}{'市值':>15}{'占比':>8}")
    print("-" * 55)
    for stock in stocks:
        percentage = (stock["value"] / total_value) * 100
        print(f"{stock['symbol']:^10}{stock['shares']:>12,.0f}${stock['price']:>11,.2f}${stock['value']:>14,.2f}{percentage:>7.1f}%")
    print("-" * 55)
    print(f"投资组合总价值: ${total_value:,.2f}")
def main():
    print("欢迎使用投资组合价值计算器")
    stocks = get_stock_info()
    print_portfolio_report(stocks)
if __name__ == "__main__":
    main()

挑战2: 金融计算器工具集

问题：创建一个综合性的金融计算器程序，包含多个功能（如复利计算、贷款计算、投资回报率计算等），并提供一个菜单界面让用户选择要使用的功能。确保程序有良好的错误处理和用户体验。

查看参考解答 
# 金融计算器工具集
def show_menu():
    print("\n=== 金融计算器工具集 ===")
    print("1. 复利计算")
    print("2. 贷款月供计算")
    print("3. 投资回报率计算")
    print("4. 通货膨胀影响计算")
    print("0. 退出程序")
    return input("请选择功能 (0-4): ")
def compound_interest():
    print("\n=== 复利计算器 ===")
    try:
        principal = float(input("请输入本金: "))
        rate = float(input("请输入年利率(例如，5%输入0.05): "))
        years = float(input("请输入投资年限: "))
        amount = principal * (1 + rate) ** years
        interest = amount - principal
        print(f"\n计算结果:")
        print(f"本金: ${principal:,.2f}")
        print(f"年利率: {rate:.2%}")
        print(f"投资年限: {years}年")
        print(f"最终金额: ${amount:,.2f}")
        print(f"利息收入: ${interest:,.2f}")
    except ValueError:
        print("错误：请输入有效的数字")
def loan_payment():
    print("\n=== 贷款月供计算器 ===")
    try:
        loan = float(input("请输入贷款金额: "))
        rate = float(input("请输入年利率(例如，5%输入0.05): "))
        years = int(input("请输入贷款年限: "))
        monthly_rate = rate / 12
        months = years * 12
        payment = loan * (monthly_rate * (1 + monthly_rate) ** months) / ((1 + monthly_rate) ** months - 1)
        total = payment * months
        interest = total - loan
        print(f"\n计算结果:")
        print(f"贷款金额: ${loan:,.2f}")
        print(f"月供金额: ${payment:,.2f}")
        print(f"总还款额: ${total:,.2f}")
        print(f"利息总额: ${interest:,.2f}")
    except ValueError:
        print("错误：请输入有效的数字")
def roi_calculator():
    print("\n=== 投资回报率计算器 ===")
    try:
        initial = float(input("请输入初始投资: "))
        final = float(input("请输入最终价值: "))
        years = float(input("请输入投资年限: "))
        total_return = final - initial
        roi = (total_return / initial) * 100
        annual_roi = ((final / initial) ** (1/years) - 1) * 100
        print(f"\n计算结果:")
        print(f"总收益: ${total_return:,.2f}")
        print(f"总回报率: {roi:.2f}%")
        print(f"年化回报率: {annual_roi:.2f}%")
    except ValueError:
        print("错误：请输入有效的数字")
    except ZeroDivisionError:
        print("错误：初始投资不能为零")
def inflation_calculator():
    print("\n=== 通货膨胀影响计算器 ===")
    try:
        amount = float(input("请输入当前金额: "))
        rate = float(input("请输入年通胀率(例如，3%输入0.03): "))
        years = int(input("请输入年数: "))
        future_value = amount / (1 + rate) ** years
        print(f"\n计算结果:")
        print(f"当前金额: ${amount:,.2f}")
        print(f"通货膨胀率: {rate:.2%}")
        print(f"{years}年后的等值购买力: ${future_value:,.2f}")
        print(f"购买力损失: ${amount - future_value:,.2f}")
    except ValueError:
        print("错误：请输入有效的数字")
def main():
    while True:
        choice = show_menu()
        if choice == '0':
            print("\n感谢使用金融计算器工具集！")
            break
        elif choice == '1':
            compound_interest()
        elif choice == '2':
            loan_payment()
        elif choice == '3':
            roi_calculator()
        elif choice == '4':
            inflation_calculator()
        else:
            print("无效的选择，请重新输入")
        input("\n按回车键继续...")
if __name__ == "__main__":
    main()
